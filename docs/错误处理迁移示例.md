# 错误处理迁移示例

本文档展示如何将现有代码迁移到统一的错误处理机制。

## 示例1: 基本错误创建

### 迁移前
```go
// internal/platforms/temu/types/errors.go
var (
    ErrProductNotFound = errors.New("产品不存在")
    ErrProductOffline  = errors.New("产品已下架")
    ErrAuthExpired     = errors.New("认证已过期")
)

// 使用
func (s *Service) GetProduct(id string) error {
    if id == "" {
        return errors.New("产品ID不能为空")
    }
    
    product, err := s.repo.Find(id)
    if err != nil {
        return fmt.Errorf("查询产品失败: %w", err)
    }
    
    if product == nil {
        return ErrProductNotFound
    }
    
    return nil
}
```

### 迁移后
```go
// 使用统一的错误包
import "task-processor/internal/core/errors"

func (s *Service) GetProduct(id string) error {
    if id == "" {
        return errors.New(errors.ErrCodeValidation, "产品ID不能为空")
    }
    
    product, err := s.repo.Find(id)
    if err != nil {
        return errors.Wrap(err, errors.ErrCodeExternalAPI, "查询产品失败")
    }
    
    if product == nil {
        return errors.New(errors.ErrCodeTaskNotFound, "产品不存在")
    }
    
    return nil
}
```

## 示例3: log.Fatal替换

### 迁移前
```go
// cmd/rabbitmq-consumer/main.go
func main() {
    logger := setupLogger()
    
    cfg, err := loadConfig()
    if err != nil {
        logger.Fatalf("加载配置失败: %v", err)
    }
    
    service, err := createService(cfg)
    if err != nil {
        logger.Fatalf("创建服务失败: %v", err)
    }
    
    if err := service.Start(); err != nil {
        logger.Fatalf("启动服务失败: %v", err)
    }
}

// internal/app/service/service.go
func (s *Service) Initialize() error {
    if err := s.loadConfig(); err != nil {
        log.Fatal("加载配置失败") // ❌ 不应该在这里使用Fatal
    }
    return nil
}
```

### 迁移后
```go
// cmd/rabbitmq-consumer/main.go
func main() {
    logger := setupLogger()
    
    if err := run(logger); err != nil {
        logger.Fatalf("程序运行失败: %v", err) // ✅ 只在main中使用Fatal
    }
}

func run(logger *logrus.Logger) error {
    cfg, err := loadConfig()
    if err != nil {
        return errors.Wrap(err, errors.ErrCodeConfig, "加载配置失败")
    }
    
    service, err := createService(cfg)
    if err != nil {
        return errors.Wrap(err, errors.ErrCodeSystem, "创建服务失败")
    }
    
    if err := service.Start(); err != nil {
        return errors.Wrap(err, errors.ErrCodeSystem, "启动服务失败")
    }
    
    return nil
}

// internal/app/service/service.go
func (s *Service) Initialize() error {
    if err := s.loadConfig(); err != nil {
        return errors.Wrap(err, errors.ErrCodeConfig, "加载配置失败") // ✅ 返回错误
    }
    return nil
}
```

## 示例4: 手动重试逻辑替换

### 迁移前
```go
// internal/platforms/temu/services/pricing/pricing_decision_service.go
func (s *PricingDecisionService) getAmazonProductWithCache(ctx context.Context, productID string) (*model.Product, error) {
    var lastErr error
    
    for attempt := 0; attempt < s.config.MaxRetries; attempt++ {
        product, err := s.productFetcher.FetchProduct(ctx, productID)
        if err == nil {
            return product, nil
        }
        
        lastErr = err
        
        // 检查是否可重试
        if !isRetryableError(err) {
            return nil, err
        }
        
        // 等待后重试
        delay := time.Duration(attempt+1) * time.Second
        time.Sleep(delay)
    }
    
    return nil, fmt.Errorf("经过%d次重试后仍无法获取产品: %w", s.config.MaxRetries, lastErr)
}
```

### 迁移后
```go
// internal/platforms/temu/services/pricing/pricing_decision_service.go
func (s *PricingDecisionService) getAmazonProductWithCache(ctx context.Context, productID string) (*model.Product, error) {
    config := errors.DefaultRetryConfig()
    config.MaxRetries = s.config.MaxRetries
    config.InitialDelay = time.Second
    config.MaxDelay = 30 * time.Second
    config.BackoffFactor = 2.0
    
    var product *model.Product
    err := errors.Retry(ctx, config, func() error {
        p, err := s.productFetcher.FetchProduct(ctx, productID)
        if err != nil {
            return errors.Wrap(err, errors.ErrCodeExternalAPI, "获取产品失败")
        }
        product = p
        return nil
    })
    
    if err != nil {
        return nil, errors.Wrapf(err, errors.ErrCodeExternalAPI, 
            "经过%d次重试后仍无法获取产品: productID=%s", config.MaxRetries, productID)
    }
    
    return product, nil
}
```

## 示例5: 错误处理器集成

### 迁移前
```go
// internal/platforms/temu/processor.go
type TemuProcessor struct {
    config *config.Config
    logger *logrus.Logger
}

func (p *TemuProcessor) ProcessTask(task Task) error {
    err := p.process(task)
    if err != nil {
        p.logger.Errorf("处理任务失败: %v", err)
        
        // 手动判断是否重试
        if isNetworkError(err) {
            return p.retryTask(task)
        }
        
        // 手动判断是否终止
        if isCriticalError(err) {
            p.logger.Error("遇到关键错误，停止处理")
            return err
        }
        
        return err
    }
    return nil
}
```

### 迁移后
```go
// internal/platforms/temu/processor.go
type TemuProcessor struct {
    config       *config.Config
    logger       *logrus.Logger
    errorHandler errors.ErrorHandler
}

func NewTemuProcessor(ctx context.Context, config *config.Config, logger *logrus.Logger) (*TemuProcessor, error) {
    if config == nil {
        return nil, errors.New(errors.ErrCodeConfig, "config不能为空")
    }
    
    return &TemuProcessor{
        config:       config,
        logger:       logger,
        errorHandler: errors.NewDefaultErrorHandler(logger),
    }, nil
}

func (p *TemuProcessor) ProcessTask(task Task) error {
    err := p.process(task)
    if err != nil {
        // 使用错误处理器统一处理
        handledErr := p.errorHandler.Handle(err)
        
        // 判断是否需要重试
        if p.errorHandler.ShouldRetry(handledErr) {
            return p.retryTask(task)
        }
        
        // 判断是否需要终止
        if p.errorHandler.ShouldTerminate(handledErr) {
            return handledErr
        }
        
        return handledErr
    }
    return nil
}
```

## 示例6: goroutine安全执行

### 迁移前
```go
// internal/app/service/config_service.go
func (s *ConfigService) StartWatching() {
    go func() {
        ticker := time.NewTicker(time.Minute)
        defer ticker.Stop()
        
        for range ticker.C {
            // 可能panic的代码
            s.reloadConfig()
        }
    }()
}
```

### 迁移后
```go
// internal/app/service/config_service.go
func (s *ConfigService) StartWatching() {
    errors.SafeGo(func() {
        ticker := time.NewTicker(time.Minute)
        defer ticker.Stop()
        
        for range ticker.C {
            // panic会被自动捕获和记录
            if err := s.reloadConfig(); err != nil {
                s.errorHandler.Handle(err)
            }
        }
    }, s.logger)
}
```

```go

    // 验证主图

    }
    
    // 验证SKU图片
```

```go
// internal/platforms/temu/utils/image_validator.go
func (v *ImageDimensionValidator) Validate(product *api.Product) error {
    var errs []error
    
    // 验证主图
    if err := v.validateMainImages(product); err != nil {
        errs = append(errs, errors.Wrap(err, errors.ErrCodeValidation, "主图尺寸验证失败"))
    }
    
    // 验证SKU图片
    if err := v.validateSkuImages(product); err != nil {
        errs = append(errs, errors.Wrap(err, errors.ErrCodeValidation, "SKU图片尺寸验证失败"))
    }
    
    // 合并所有错误
    return errors.Combine(errs...)
}
```

## 示例8: 自定义错误类型迁移

### 迁移前
```go
// internal/platforms/temu/types/errors.go
type RetryableError struct {
    Message string
    Cause   error
}

func (e *RetryableError) Error() string {
    if e.Cause != nil {
        return fmt.Sprintf("%s: %v", e.Message, e.Cause)
    }
    return e.Message
}

func NewRetryableError(message string, cause error) *RetryableError {
    return &RetryableError{
        Message: message,
        Cause:   cause,
    }
}

// 使用
func (s *Service) FetchData() error {
    data, err := s.api.Get()
    if err != nil {
        return NewRetryableError("获取数据失败", err)
    }
    return nil
}
```

### 迁移后
```go
// 使用统一的错误包，不需要自定义类型
func (s *Service) FetchData() error {
    data, err := s.api.Get()
    if err != nil {
        // 使用可重试的错误码
        return errors.Wrap(err, errors.ErrCodeNetwork, "获取数据失败")
    }
    return nil
}

// 判断是否可重试
if errors.IsRetryable(err) {
    // 重试逻辑
}
```

## 迁移步骤总结

### 第一阶段：基础迁移
1. ✅ 替换 `errors.New` → `errors.New(code, message)`
2. ✅ 替换 `fmt.Errorf` → `errors.Wrap/Wrapf`
3. ✅ 移除非main函数的 `log.Fatal`

### 第二阶段：功能增强
4. ✅ 构造函数返回 `(value, error)`
5. ✅ 使用 `errors.Retry` 替换手动重试
6. ✅ 使用 `errors.SafeGo` 启动goroutine

### 第三阶段：架构优化
7. ✅ 集成错误处理器
8. ✅ 统一错误日志记录
9. ✅ 添加错误监控和告警
