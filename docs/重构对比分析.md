```go
// 独立的工作池
type WorkerPool struct {
    maxWorkers int
    taskChan   chan *ProcessTask
    resultChan chan *ProcessResult
    wg         sync.WaitGroup
    logger     *logrus.Entry
}

func (wp *WorkerPool) Start(ctx context.Context, processFunc ProcessFunc)
func (wp *WorkerPool) Submit(task *ProcessTask)
func (wp *WorkerPool) Close()
func (wp *WorkerPool) Wait()
func (wp *WorkerPool) Results() <-chan *ProcessResult

// 简化的并行处理器
type ParallelProcessor struct {
    maxWorkers int
    timeout    time.Duration
    logger     *logrus.Entry
}

func (pp *ParallelProcessor) ProcessParallel(...) []*ProcessResult {
    // 1. 创建工作池 (3行)
    pool := NewWorkerPool(workerCount, len(tasks), pp.logger)
    pool.Start(ctx, pp.wrapProcessFunc(processFunc))
    
    // 2. 分发任务 (3行)
    for _, task := range tasks {
        pool.Submit(task)
    }
    pool.Close()
    
    // 3. 收集结果 (5行)
    go pool.Wait()
    results := make([]*ProcessResult, 0, len(tasks))
    for result := range pool.Results() {
        results = append(results, result)
    }
    
    // 4. 统计结果 (1行)
    pp.logStatistics(results, len(tasks), workerCount)
}
```

**改进**:
- ✅ 工作池独立管理
- ✅ 职责清晰分离
- ✅ 易于单独测试
- ✅ 代码更简洁
