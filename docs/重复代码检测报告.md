#### 5. Amazon URL解析重复 (44行重复)

- 第63-106行：`GetCurrencyFromURL`
- 第109-152行：`GetMarketplaceFromURL`

**重复内容：** 完全相同的域名解析switch语句，仅返回值不同

**重构建议：**
```go
// 域名到市场信息的映射
var marketplaceInfo = map[string]struct{
    marketplace string
    currency    string
}{
    "amazon.com":    {"US", "USD"},
    "amazon.co.uk":  {"UK", "GBP"},
    "amazon.de":     {"DE", "EUR"},
    // ... 其他市场
}

func (uh *URLHelper) getMarketplaceInfo(productURL string) (marketplace, currency string) {
    parsedURL, err := url.Parse(productURL)
    if err != nil {
        return "US", "USD"
    }
    
    host := strings.ToLower(parsedURL.Host)
    for domain, info := range marketplaceInfo {
        if strings.Contains(host, domain) {
            return info.marketplace, info.currency
        }
    }
    return "US", "USD"
}

func (uh *URLHelper) GetCurrencyFromURL(productURL string) string {
    _, currency := uh.getMarketplaceInfo(productURL)
    return currency
}

func (uh *URLHelper) GetMarketplaceFromURL(productURL string) string {
    marketplace, _ := uh.getMarketplaceInfo(productURL)
    return marketplace
}
```

func (s *SubmitAPI) temuRequest(
    endpoint string,
    request interface{},
    result interface{},
    actionName string,
) error {
    headers := s.buildStandardHeaders()
    
    apiReq := map[string]interface{}{
        "method":  "POST",
        "url":     endpoint,
        "headers": headers,
        "body":    request,
    }
    
    if err := s.client.SendTEMURequest(apiReq, result); err != nil {
        s.logger.WithError(err).Errorf("%s请求失败", actionName)
        return fmt.Errorf("%s请求失败: %w", actionName, err)
    }
    
    return s.validateResponse(result, actionName)
}

    headers := client.GetDefaultHeaders()
    headers["accept"] = "application/json, text/plain, */*"
    headers["content-type"] = "application/json;charset=UTF-8"
    // ... 其他标准头

func (v *SkuImageValidator) validateImages(
    images []models.ImageInfo,
    skcIndex, skuIndex int,
    requirement services.ImageRequirement,
    paddedImagesMap map[string][]byte,
    paddedSizesMap map[string][2]int,

## 附录：检测工具安装

### golangci-lint安装

```bash
# Windows (使用Go安装)
go install github.com/golangci/golangci-lint/cmd/golangci-lint@latest

# macOS
brew install golangci-lint

# Linux
curl -sSfL https://raw.githubusercontent.com/golangci/golangci-lint/master/install.sh | sh -s -- -b $(go env GOPATH)/bin
```

### 运行检测

```bash
# 检测重复代码
golangci-lint run --enable dupl ./...

# 调整阈值（默认50行）
golangci-lint run --enable dupl --dupl-threshold=30 ./...

# 生成JSON报告
golangci-lint run --enable dupl --out-format=json > dupl_report.json
```
