## ✍️ 编写测试

### 1. 单元测试

#### 基本结构
```go
package mypackage

import (
    "testing"
    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"
)

func TestMyFunction(t *testing.T) {
    // 准备 (Arrange)
    input := "测试输入"
    expected := "期望输出"
    
    // 执行 (Act)
    result := MyFunction(input)
    
    // 断言 (Assert)
    assert.Equal(t, expected, result)
}
```

#### 表驱动测试
```go
func TestMyFunction(t *testing.T) {
    tests := []struct {
        name     string
        input    string
        expected string
        wantErr  bool
    }{
        {
            name:     "有效输入",
            input:    "test",
            expected: "TEST",
            wantErr:  false,
        },
        {
            name:     "空输入",
            input:    "",
            expected: "",
            wantErr:  true,
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            result, err := MyFunction(tt.input)
            
            if tt.wantErr {
                assert.Error(t, err)
            } else {
                assert.NoError(t, err)
                assert.Equal(t, tt.expected, result)
            }
        })
    }
}
```

### 2. Mock对象

#### 使用接口
```go
// 定义接口
type DataStore interface {
    Get(key string) (string, error)
    Set(key, value string) error
}

// Mock实现
type MockDataStore struct {
    data map[string]string
}

func (m *MockDataStore) Get(key string) (string, error) {
    if val, ok := m.data[key]; ok {
        return val, nil
    }
    return "", errors.New("未找到")
}

func (m *MockDataStore) Set(key, value string) error {
    m.data[key] = value
    return nil
}

// 使用mock进行测试
func TestMyService(t *testing.T) {
    mock := &MockDataStore{
        data: map[string]string{"key1": "value1"},
    }
    
    service := NewMyService(mock)
    result, err := service.Process("key1")
    
    assert.NoError(t, err)
    assert.Equal(t, "value1", result)
}
```

### 3. 测试错误处理

```go
func TestErrorHandling(t *testing.T) {
    t.Run("无效输入返回错误", func(t *testing.T) {
        _, err := ProcessData(nil)
        
        require.Error(t, err)
        assert.Contains(t, err.Error(), "无效输入")
    })
    
    t.Run("包装底层错误", func(t *testing.T) {
        _, err := ProcessData(invalidData)
        
        require.Error(t, err)
        assert.True(t, errors.Is(err, ErrValidation))
    })
}
```

### 4. 测试并发操作

```go
func TestConcurrentOperation(t *testing.T) {
    done := make(chan bool)
    
    go func() {
        defer close(done)
        // 并发操作
        result := DoWork()
        assert.NotNil(t, result)
    }()
    
    select {
    case <-done:
        // 成功
    case <-time.After(5 * time.Second):
        t.Fatal("操作超时")
    }
}
```

### 5. 基准测试

```go
func BenchmarkMyFunction(b *testing.B) {
    input := "测试输入"
    
    b.ResetTimer()
    for i := 0; i < b.N; i++ {
        MyFunction(input)
    }
}

func BenchmarkMyFunctionParallel(b *testing.B) {
    input := "测试输入"
    
    b.RunParallel(func(pb *testing.PB) {
        for pb.Next() {
            MyFunction(input)
        }
    })
}
```

## 🚀 运行测试

### 基本命令

```powershell
# 运行所有测试
go test ./...

# 详细输出
go test -v ./...

# 运行特定包的测试
go test ./internal/core/errors/...

# 运行特定测试
go test -run TestMyFunction ./...

# 运行测试并显示覆盖率
go test -cover ./...

go test -coverprofile=coverage.out ./...
```

## 📝 测试最佳实践

### ✅ 推荐做法

1. **先写测试 (TDD)**
   ```go
   // 先写测试
   func TestNewFeature(t *testing.T) {
       result := NewFeature()
       assert.NotNil(t, result)
   }
   
   // 再实现功能
   func NewFeature() *Feature {
       return &Feature{}
   }
   ```

2. **使用表驱动测试**
   - 更容易添加新测试用例
   - 更好的测试覆盖
   - 更清晰的测试意图

3. **测试错误情况**
   ```go
   func TestErrorCases(t *testing.T) {
       tests := []struct {
           name    string
           input   interface{}
           wantErr bool
       }{
           {"nil输入", nil, true},
           {"无效类型", "string", true},
           {"有效输入", 123, false},
       }
       // ...
   }
   ```

4. **使用有意义的测试名称**
   ```go
   func TestUserService_CreateUser_有效数据_返回用户(t *testing.T)
   func TestUserService_CreateUser_重复邮箱_返回错误(t *testing.T)
   ```

5. **清理资源**
   ```go
   func TestWithCleanup(t *testing.T) {
       file, err := os.CreateTemp("", "test")
       require.NoError(t, err)
       defer os.Remove(file.Name())
       
       // 测试代码
   }
   ```

6. **使用子测试组织代码**
   ```go
   func TestMyFunction(t *testing.T) {
       t.Run("有效输入", func(t *testing.T) {
           // 测试有效输入
       })
       
       t.Run("无效输入", func(t *testing.T) {
           // 测试无效输入
       })
   }
   ```

### ❌ 避免做法

1. **不要测试实现细节**
   ```go
   // 不好 - 测试内部状态
   func TestBad(t *testing.T) {
       obj := NewObject()
       assert.Equal(t, 0, obj.internalCounter)
   }
   
   // 好 - 测试行为
   func TestGood(t *testing.T) {
       obj := NewObject()
       result := obj.Process()
       assert.Equal(t, expected, result)
   }
   ```

2. **不要使用sleep进行时序控制**
   ```go
   // 不好
   go doWork()
   time.Sleep(1 * time.Second)
   
   // 好
   done := make(chan bool)
   go func() {
       doWork()
       done <- true
   }()
   select {
   case <-done:
   case <-time.After(5 * time.Second):
       t.Fatal("超时")
   }
   ```

3. **不要忽略测试中的错误**
   ```go
   // 不好
   result, _ := MyFunction()
   
   // 好
   result, err := MyFunction()
   require.NoError(t, err)
   ```

4. **不要使用全局状态**
   ```go
   // 不好
   var globalConfig *Config
   
   func TestBad(t *testing.T) {
       globalConfig = &Config{}
       // 测试可能相互干扰
   }
   
   // 好
   func TestGood(t *testing.T) {
       config := &Config{}
       service := NewService(config)
       // 每个测试都是隔离的
   }
   ```

## 🔗 集成测试

### 结构
```go
func TestIntegration(t *testing.T) {
    if testing.Short() {
        t.Skip("跳过集成测试")
    }
    
    // 设置
    db := setupTestDB(t)
    defer db.Close()
    
    // 测试
    result := FullWorkflow(db)
    
    // 验证
    assert.NotNil(t, result)
}
```

### 运行集成测试
```powershell
# 跳过集成测试
go test -short ./...

# 只运行集成测试
go test -run Integration ./...
```
